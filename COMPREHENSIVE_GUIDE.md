### Multi-Locale App
```typescript
const locales = ['us', 'jp', 'de', 'br', 'in', 'sg', 'ae'];
const users = locales.map(locale => {
  const m = createMimic(locale);
  return {
    locale,
    name: m.identity.fullName(),
    address: m.location.fullAddress()
  };
});
```
